#include <tunables/global>

profile supertonic_tts flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities needed for network binding and process management
  capability net_bind_service,
  capability setgid,
  capability setuid,

  # S6-Overlay supervision system
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio helper library
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Add-on configuration and data
  /data/** rw,

  # Shared directory access (for potential future features)
  /share/** rw,

  # Start new profile for Python service
  /usr/bin/python3 cx -> python3,

  profile python3 flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>
    #include <abstractions/python>

    # Suppress ptrace denials (not needed)
    deny ptrace (read),

    # Python interpreter and standard library
    /usr/bin/python3* mr,
    /usr/lib/python3*/** mr,

    # Python virtual environment and packages
    /opt/venv/** mr,
    /opt/venv/lib/python3*/** mr,

    # Supertonic models and helper code (read-only)
    /opt/supertonic/** r,
    /opt/supertonic/models/** r,
    /opt/supertonic/py/** r,

    # Temporary files for audio generation
    /tmp/** rw,
    /dev/shm/** rw,

    # Network access for Wyoming protocol server
    network inet stream,
    network inet6 stream,
    network inet dgram,
    network inet6 dgram,

    # Add-on data directory
    /data/** rw,

    # Shared data directory
    /share/** rw,

    # Python script execution
    /usr/bin/supertonic_server.py r,

    # System configuration files (read-only)
    /etc/hosts r,
    /etc/resolv.conf r,
    /etc/nsswitch.conf r,
    /etc/services r,
    /etc/passwd r,
    /etc/group r,

    # Proc filesystem for process information
    /proc/ r,
    /proc/sys/kernel/hostname r,
    /proc/sys/kernel/osrelease r,
    /proc/*/stat r,
    /proc/*/status r,
    /proc/*/cmdline r,
    /proc/sys/kernel/random/** r,

    # Device files for random number generation
    /dev/random r,
    /dev/urandom r,
    /dev/null rw,
    /dev/zero rw,
    /dev/full rw,

    # Allow reading CPU info for ONNX optimizations
    /sys/devices/system/cpu/** r,
  }
}
